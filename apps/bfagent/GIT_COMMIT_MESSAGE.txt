**Model Changes:**

category_fk = ForeignKey('HandlerCategory', on_delete=PROTECT, null=True)

# Kept old field (deprecated)
category = CharField(max_length=20, null=True)  

# Helper properties for compatibility
@property
def category_code(self):
    return self.category_fk.code if self.category_fk else self.category
