#!/usr/bin/env bash
set -euo pipefail

PKG="docs-publish-kit"
ARCHIVE="${PKG}.tar.gz"

rm -rf "${PKG}" "${ARCHIVE}"
mkdir -p "${PKG}"

# -----------------------------
# README
# -----------------------------
cat > "${PKG}/README.md" <<'EOF'
# Docs Publish Kit (Sphinx + GitHub Actions + Hetzner + Docker + Host-Nginx)

Dieses Paket liefert:
- GitHub Actions: PR-Gate (sphinx -W + linkcheck) + Deploy (main + tags v*)
- Hetzner: immutable releases + atomarer "current" symlink + retention
- Docker: nginx:alpine Container serviert /var/www/docs/current (read-only)
- Host-Nginx vHost Template für Subdomain + TLS + sinnvolle Cache-Header
- Empfohlenes Sphinx conf.py (Python/Django): autoapi, napoleon, intersphinx, myst, opengraph, nitpicky

## Quickstart

### 1) Dateien ins Repo übernehmen
Kopiere:
- `.github/workflows/docs.yml`
- `docs/` (requirements, source, conf.py, index.md, templates)
- optional `server/` (compose + nginx template) auf den Hetzner Host

### 2) GitHub Secrets setzen
Repo → Settings → Secrets and variables → Actions:
- HETZNER_HOST
- HETZNER_USER  (z. B. deploy)
- HETZNER_SSH_KEY (private key)
- optional HETZNER_PORT

### 3) Hetzner vorbereiten
- `/var/www/docs/{releases,versions}` anlegen
- deploy-user und Rechte setzen (siehe server/SETUP_SERVER.md)
- `docker compose up -d` in `server/`

### 4) Host-Nginx vHost aktivieren
- `server/nginx/docs.example.tld.conf` nach `/etc/nginx/sites-available/` kopieren
- Subdomain + Zertifikat-Pfade anpassen
- `nginx -t && systemctl reload nginx`

### 5) Branch Protection
Für `main`:
- Require status checks → Workflow "Docs / build"

## Rollback
Auf dem Server:
- `ls -1dt /var/www/docs/releases/* | head`
- `ln -sfn /var/www/docs/releases/<OLD_SHA> /var/www/docs/current`

## Hinweise für Django
- Dieses Setup vermeidet bewusst ein `django.setup()` im Doc-Build (robuster).
- Falls du Django-Imports im autodoc brauchst, nutze `autodoc_mock_imports` oder setze ENV in CI.
EOF

# -----------------------------
# GitHub Actions Workflow
# -----------------------------
mkdir -p "${PKG}/.github/workflows"
cat > "${PKG}/.github/workflows/docs.yml" <<'EOF'
name: Docs

on:
  pull_request:
  push:
    branches: ["main"]
    tags: ["v*"]
  workflow_dispatch:

permissions:
  contents: read

concurrency:
  group: docs-${{ github.ref }}
  cancel-in-progress: true

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.12"

      - name: Install docs deps
        run: |
          set -euo pipefail
          python -m pip install --upgrade pip
          python -m pip install -r docs/requirements.txt

      - name: Build HTML (warnings as errors)
        run: |
          set -euo pipefail
          sphinx-build -W -b html docs/source docs/_build/html

      - name: Linkcheck
        run: |
          set -euo pipefail
          sphinx-build -b linkcheck docs/source docs/_build/linkcheck

      - name: Write build info
        run: |
          set -euo pipefail
          {
            echo "ref=${GITHUB_REF}"
            echo "sha=${GITHUB_SHA}"
            echo "built_at_utc=$(date -u +'%Y-%m-%dT%H:%M:%SZ')"
          } > docs/_build/html/build-info.txt

      - name: Upload artifact
        uses: actions/upload-artifact@v4
        with:
          name: sphinx-html
          path: docs/_build/html

  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.event_name == 'push' && (startsWith(github.ref, 'refs/heads/main') || startsWith(github.ref, 'refs/tags/v'))
    steps:
      - name: Download artifact
        uses: actions/download-artifact@v4
        with:
          name: sphinx-html
          path: html

      - name: Install SSH key
        uses: webfactory/ssh-agent@v0.9.0
        with:
          ssh-private-key: ${{ secrets.HETZNER_SSH_KEY }}

      - name: Add host key
        run: |
          set -euo pipefail
          mkdir -p ~/.ssh
          ssh-keyscan -H "${{ secrets.HETZNER_HOST }}" >> ~/.ssh/known_hosts

      - name: Deploy (atomic release + switch)
        env:
          HOST: ${{ secrets.HETZNER_HOST }}
          USER: ${{ secrets.HETZNER_USER }}
          PORT: ${{ secrets.HETZNER_PORT }}
          SHA: ${{ github.sha }}
          REF: ${{ github.ref }}
        run: |
          set -euo pipefail

          SSH_PORT="${PORT:-22}"
          SSH="ssh -p ${SSH_PORT} ${USER}@${HOST}"

          if [[ "${REF}" == "refs/heads/main" ]]; then
            TARGET_DIR="/var/www/docs/releases/${SHA}"
            LINK="/var/www/docs/current"
          else
            TAG="${REF#refs/tags/}"
            TARGET_DIR="/var/www/docs/versions/${TAG}"
            LINK=""
          fi

          ${SSH} "mkdir -p '${TARGET_DIR}'"
          rsync -az --delete -e "ssh -p ${SSH_PORT}" html/ "${USER}@${HOST}:${TARGET_DIR}/"

          if [[ -n "${LINK}" ]]; then
            ${SSH} "ln -sfn '${TARGET_DIR}' '${LINK}'"
            # Retention: keep last 20 releases
            ${SSH} "ls -1dt /var/www/docs/releases/* | tail -n +21 | xargs -r rm -rf"
          fi
EOF

# -----------------------------
# Server: Docker compose + Nginx vhost template + setup notes
# -----------------------------
mkdir -p "${PKG}/server/nginx"

cat > "${PKG}/server/docker-compose.yml" <<'EOF'
services:
  docs:
    image: nginx:1.27-alpine
    container_name: docs-static
    restart: unless-stopped
    ports:
      - "127.0.0.1:8081:80"
    volumes:
      - /var/www/docs/current:/usr/share/nginx/html:ro
    healthcheck:
      test: ["CMD-SHELL", "wget -qO- http://localhost/ | grep -qi '<html'"]
      interval: 30s
      timeout: 5s
      retries: 3
EOF

cat > "${PKG}/server/nginx/docs.example.tld.conf" <<'EOF'
# Template: /etc/nginx/sites-available/docs.example.tld
# Replace:
# - docs.example.tld
# - ssl_certificate paths

server {
    listen 80;
    server_name docs.example.tld;

    location /.well-known/acme-challenge/ {
        root /var/www/letsencrypt;
    }

    location / {
        return 301 https://$host$request_uri;
    }
}

server {
    listen 443 ssl http2;
    server_name docs.example.tld;

    ssl_certificate     /etc/letsencrypt/live/docs.example.tld/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/docs.example.tld/privkey.pem;

    add_header X-Content-Type-Options nosniff always;
    add_header X-Frame-Options DENY always;
    add_header Referrer-Policy strict-origin-when-cross-origin always;

    # Enable only once TLS is stable:
    # add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    location / {
        proxy_pass http://127.0.0.1:8081;
        proxy_http_version 1.1;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # HTML should be fresh
        add_header Cache-Control "no-cache" always;
    }

    location ~* \.(?:css|js|jpg|jpeg|png|gif|svg|ico|woff2|woff|ttf)$ {
        proxy_pass http://127.0.0.1:8081;
        expires 30d;
        add_header Cache-Control "public, max-age=2592000, immutable" always;
    }
}
EOF

cat > "${PKG}/server/SETUP_SERVER.md" <<'EOF'
# Server Setup (Hetzner)

## 1) Directories
sudo mkdir -p /var/www/docs/{releases,versions}
sudo mkdir -p /var/www/letsencrypt

## 2) Deploy user (recommended)
sudo adduser --disabled-password --gecos "" deploy
sudo usermod -aG docker deploy
sudo chown -R deploy:deploy /var/www/docs

## 3) Start docs container
sudo mkdir -p /opt/docs
sudo cp docker-compose.yml /opt/docs/docker-compose.yml
cd /opt/docs
sudo docker compose up -d

## 4) Host nginx vhost
sudo cp nginx/docs.example.tld.conf /etc/nginx/sites-available/docs.example.tld
# edit domain + cert paths
sudo ln -s /etc/nginx/sites-available/docs.example.tld /etc/nginx/sites-enabled/docs.example.tld
sudo nginx -t
sudo systemctl reload nginx

## 5) Test
curl -I https://docs.example.tld/
curl -s https://docs.example.tld/build-info.txt | head
EOF

from future import annotations

import os
import sys
from datetime import datetime
from typing import Any

-- Path setup --------------------------------------------------------------
Assumption: repo layout has project package in root or in src/
You may need to adjust these two lines.
REPO_ROOT = os.path.abspath(os.path.join(os.path.dirname(file), "../.."))
SRC_ROOT = os.path.join(REPO_ROOT, "src")

sys.path.insert(0, REPO_ROOT)
if os.path.isdir(SRC_ROOT):
sys.path.insert(0, SRC_ROOT)

-- Project information -----------------------------------------------------
project = os.environ.get("DOCS_PROJECT_NAME", "My Project")
author = os.environ.get("DOCS_AUTHOR", "Team")
copyright = f"{datetime.utcnow().year}, {author}"

Prefer version from env (e.g. tag). Keep deterministic for CI.
release = os.environ.get("DOCS_RELEASE", os.environ.get("GITHUB_REF_NAME", "dev"))
version = release

-- General configuration ---------------------------------------------------
extensions = [
# Core
"sphinx.ext.autodoc",
"sphinx.ext.napoleon",
"sphinx.ext.intersphinx",
"sphinx.ext.viewcode",
"sphinx.ext.autosectionlabel",
# Markdown
"myst_parser",
# Auto-generate API pages from source
"autoapi.extension",
# Social previews
"sphinxext.opengraph",
]

templates_path = ["_templates"]
exclude_patterns = ["_build", "Thumbs.db", ".DS_Store"]

language = "de"

-- MyST --------------------------------------------------------------------
myst_enable_extensions = [
"deflist",
"fieldlist",
"colon_fence",
"substitution",
"tasklist",
]
myst_heading_anchors = 3

-- Theme -------------------------------------------------------------------
html_theme = "furo"
html_static_path = ["_static"]

(Optional) add a small footer note with build-info, if you want:
html_theme_options: dict[str, Any] = {
"footer_icons": [],
}

-- Napoleon (Docstring styles) --------------------------------------------
napoleon_google_docstring = True
napoleon_numpy_docstring = True
napoleon_use_param = True
napoleon_use_rtype = True
napoleon_preprocess_types = True

-- Intersphinx -------------------------------------------------------------
intersphinx_mapping = {
"python": ("https://docs.python.org/3", {}),
"django": ("https://docs.djangoproject.com/en/stable/", "https://docs.djangoproject.com/en/stable/_objects/"),
}

-- AutoAPI -----------------------------------------------------------------
IMPORTANT:
Use AutoAPI to avoid importing Django / running django.setup() during docs build.
This keeps builds reproducible and prevents CI failures due to missing env/settings.
autoapi_type = "python"
autoapi_keep_files = False
autoapi_add_toctree_entry = False

Choose ONE of the two below:
1) If you have src/ layout, keep as-is:

What to include

"members",
"undoc-members",
"show-inheritance",
"show-module-summary",
"imported-members",
]

Exclude typical noisy paths

]

mock imports here (keeps docs build from needing Django settings):

"rest_framework",
]

autodoc_typehints = "description"
autodoc_member_order = "bysource"

-- Nitpicky mode (treat missing refs as problems) --------------------------

Ignore a few common dynamic refs (tune to your project)
nitpick_ignore = [
("py:class", "QuerySet"),
("py:class", "Manager"),
("py:class", "HttpRequest"),
("py:class", "HttpResponse"),
]

-- Autosectionlabel --------------------------------------------------------
autosectionlabel_prefix_document = True

-- OpenGraph ---------------------------------------------------------------
ogp_site_url = os.environ.get("DOCS_SITE_URL", "https://docs.example.tld/")
ogp_use_first_image = True
ogp_image = os.environ.get("DOCS_OG_IMAGE", "https://docs.example.tld/_static/og.png")
ogp_description_length = 200

-- Build hygiene -----------------------------------------------------------
Keep warnings visible; CI uses -W already.
suppress_warnings = []

Default role for inline code-ish refs (optional)
default_role = "any"
EOF

## So benutzt du das Paket

1) Speichere den Block oben z. B. als `make-docs-kit.sh`  
2) Ausführen:
```bash
bash make-docs-kit.sh
Du bekommst:

docs-publish-kit/ (Ordner)

docs-publish-kit.tar.gz (Archiv = dein „Download“-Paket)

Was du danach noch anpassen musst (bewusst minimal)
In server/nginx/docs.example.tld.conf: Subdomain + Zertifikatpfade

In docs/source/conf.py: falls du ein echtes src/ Layout hast, passt es schon; sonst ggf. autoapi_dirs auf deinen Paketpfad setzen

GitHub Secrets setzen (Host/User/Key)

Wenn du mir kurz sagst, ob dein Django-Code unter src/ liegt oder z. B. im Repo-Root (z. B. project/), kann ich dir die AutoAPI-Pfade exakt auf dein Layout vorkonfigurieren (ohne Rateversuche in CI).
